2/8: Inverted Half Pyramid N = int(input()) for line_number in range(N): pattern = "" for i in range(1, N - line_number + 1): pattern = pattern + str(i) + " " print(pattern) https://t.co/e3u43vp3Qi

3/8: Number in Square Pattern N = int(input()) number = 1 for i in range(N): pattern = "" for j in range(N): pattern = pattern + str(number) + " " number = number + 1 print(pattern) https://t.co/qAlePsaMHG

4/8: Half Pyramid - 2 starting_num = int(input()) row = int(input()) for i in range(1, row+1): pattern = "" for j in range(1, i+1): pattern = pattern + str(starting_num) + " " starting_num = starting_num + 1 print(pattern) https://t.co/TBO4OFUrGm

5/8: Hollow Pyramid - 1 row = int(input()) for line_num in range(1, row+1): if line_num > 1 and line_num < row: print("* "+ " " * (line_num - 2) + "* ") else: print("* " * line_num) https://t.co/PEPqKhjekc

6/8: Full Pyramid - 1 n = int(input()) for i in range(1, n+1): zero = "0 " * (n - i) one = "1 " * ((2 * i) - 1) print(zero + one + zero) https://t.co/R6XUyvVUXO

7/8: Full Pyramid - 2 N = int(input()) for i in range(1, N+1): pattern = "" space = " " * (N - i) for j in range(1, i+1): pattern = pattern + str(j) + " " print(space + pattern)

8/8: Inverted Full Pyramid - 2 row = int(input()) for i in range(0, row+1): pattern = "" space = " " * i for number in range(1, row+1-i): pattern = pattern + str(number) + " " print(space + pattern) https://t.co/9HPkxwBZJP
